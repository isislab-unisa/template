\chapter{Introduction}
\section{Research context}
Cyber-physical systems combine sensing, computation, networking, and actuation. Multimodal learning can improve situational awareness, but it also raises questions about robustness, interpretability, privacy, and reproducibility \cite{goodfellow2016deep,nist2023airmf}.

\begin{researchquestion}[RQ1 -- Robustness]
How can a multimodal model remain reliable when one sensor stream is noisy, delayed, or unavailable?
\end{researchquestion}

\begin{researchquestion}[RQ2 -- Trust]
Which design and evaluation choices make system behaviour more transparent to operators and auditors?
\end{researchquestion}

\section{Contributions}
\begin{keypoint}[Contributions]
\begin{enumerate}[label=\textcolor{ISISBlue}{\sffamily\bfseries C\arabic*.},leftmargin=2.2em]
  \item A modular fusion architecture for heterogeneous sensor streams.
  \item A reliability-aware training objective and missing-modality protocol.
  \item A reproducible evaluation suite covering accuracy, calibration, latency, and robustness.
\end{enumerate}
\end{keypoint}
