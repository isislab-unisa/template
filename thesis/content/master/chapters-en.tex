\chapter{Introduction}
\section{Research context}
Cyber-physical systems combine sensing, computation, communication, and actuation. Multimodal learning can improve situational awareness, but it also introduces questions of robustness, explainability, privacy, and reproducibility \cite{goodfellow2016deep,nist2023airmf}.

\begin{researchquestion}[RQ1 -- Robustness]
How can a multimodal model remain reliable when one sensor stream is noisy, delayed, or unavailable?
\end{researchquestion}

\begin{researchquestion}[RQ2 -- Trust]
Which design and evaluation choices make system behaviour more transparent to operators and auditors?
\end{researchquestion}

\section{Contributions}
\begin{keypoint}[Contributions]
\begin{enumerate}
  \item A modular fusion architecture for heterogeneous sensor streams.
  \item A reliability-aware objective and a protocol for missing modalities.
  \item A reproducible evaluation suite for accuracy, calibration, latency, and robustness.
\end{enumerate}
\end{keypoint}
