\subsection{Problem statement}
\begin{frame}{Problem and contributions}
\begin{columns}[T,onlytextwidth]
\begin{column}{.48\textwidth}
\ISISCard[ISISRed][.95\linewidth]{Problem}{%
Existing systems are accurate in controlled settings, but performance degrades when the data distribution changes. This creates unreliable outcomes in real deployments.
}
\vspace{3mm}
\ISISCard[ISISLogoBlue][.95\linewidth]{Research question}{%
Can we design a method that remains accurate, efficient, and interpretable under realistic distribution shifts?
}
\end{column}
\begin{column}{.48\textwidth}
\ISISCard[ISISGreen][.95\linewidth]{Contributions}{%
\ISISVisualList[ISISGreen]{A new robust learning formulation, A reproducible experimental protocol, Evidence across three benchmark scenarios}%
}
\end{column}
\end{columns}
\end{frame}

\section{Related Work and Gap}
\ISISSectionIntro[02]{Related Work and Gap}[Position the contribution without turning the talk into a bibliography][ISISCyan]{%
  \ISISBadge[ISISCyan]{Positioning}\quad Compare families of approaches around the unresolved gap.
}
\subsection{Positioning}
\begin{frame}{Where the paper fits}
\begin{columns}[T,onlytextwidth]
\begin{column}{.36\textwidth}
\vspace{4mm}
\ISISMetricCard[ISISLogoBlue][.78\linewidth]{12}{core papers}\vspace{8mm}
\ISISMetricCard[ISISViolet][.78\linewidth]{3}{main families}
\end{column}
\begin{column}{.60\textwidth}
\ISISCard[ISISLogoBlue][.95\linewidth]{Comparison logic}{%
Avoid listing related work too early. First explain the problem and the idea; then compare alternatives around the gap the paper addresses.
}
\vspace{3mm}
\ISISCard[ISISViolet][.95\linewidth]{Gap statement}{%
Prior work optimizes either accuracy or robustness, but does not jointly report reliability, computational cost, and interpretability.
}
\end{column}
\end{columns}
\end{frame}

\subsection{Evidence plan}
\begin{frame}{Claims must map to evidence}
\centering
\ISISPipeline[3]{Claim/What we promise, Method/How we test, Evidence/Data and analysis, Interpretation/What it means}
\vspace{5mm}
\ISISCard[ISISGreen][.76\linewidth]{Evidence checklist}{%
Every claim introduced in the first part of the presentation should be revisited with a figure, table, proof sketch, or experiment in the results section.
}
\end{frame}

\section{Method}
\ISISSectionIntro[03]{Method}[Move from intuition to a reproducible algorithm][ISISGreen]{%
  \ISISBadge[ISISGreen]{Core idea}\quad Explain inputs, transformations, assumptions, and outputs.
}
\subsection{Overview}
\begin{frame}{Method pipeline}
\centering
\ISISPipeline[2]{Input/Raw data, Model/Proposed method, Validation/Metrics, Release/Code and data}
\vspace{5mm}
\begin{columns}[T,onlytextwidth]
\begin{column}{.47\textwidth}
\ISISCard[ISISLogoBlue][.96\linewidth]{Design principle}{%
Separate the core idea from implementation details. Show the intuition first, then the formal or algorithmic specification.
}
\end{column}
\begin{column}{.47\textwidth}
\ISISCard[ISISRed][.96\linewidth]{Method slide rule}{%
One pipeline slide should explain how data becomes evidence. Later slides can zoom into each module.
}

\end{columns}
\end{frame}
