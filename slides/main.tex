\begin{frame}{Problem and contributions}
\vspace*{-1truecm}% manual positioning for dense slides
\begin{columns}[T,onlytextwidth]
\begin{column}{.48\textwidth}
\ISISCard[ISISRed][.95\linewidth]{Problem}{%
Existing systems are accurate in controlled settings, but performance degrades when the data distribution changes. This creates unreliable outcomes in real deployments.
}
\medskip

\ISISCard[ISISLogoBlue][.95\linewidth]{Research question}{%
